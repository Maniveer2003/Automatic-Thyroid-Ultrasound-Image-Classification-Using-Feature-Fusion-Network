% --- Executes on button press in pushbutton3.
function pushbutton3_Callback(hObject, eventdata, handles)
% hObject    handle to pushbutton3 (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)
global img; global Thyroid;global I3;
s=I3;
num_iter = 10;
    delta_t = 1/7;
    kappa = 15;
    option = 2;
    inp = anisodiff(s,num_iter,delta_t,kappa,option);
    inp = uint8(inp);
    
inp=imresize(inp,[256,256]);
if size(inp,3)>1
   inp=rgb2gray(inp);
end
I3=inp;
sout=imresize(inp,[256,256]);
t0=60;
th=t0+((max(inp(:))+min(inp(:)))./2);
for i=1:1:size(inp,1)
    for j=1:1:size(inp,2)
        if inp(i,j)>th
            sout(i,j)=1;
        else
            sout(i,j)=0;
        end
    end
end
label=bwlabel(sout);
stats=regionprops(logical(sout),'Solidity','Area','BoundingBox');
density=[stats.Solidity];
area=[stats.Area];
high_dense_area=density>0.1;
max_area=max(area(high_dense_area));
Thyroid_label=find(area==max_area);
Thyroid=ismember(label,Thyroid_label);

if max_area>100
else
msgbox('No Thyroid!!','status');
axes(handles.axes1); cla(handles.axes1); title(''); axis off
    return;
end
box = stats(Thyroid_label);
wantedBox = box.BoundingBox;
dilationAmount = 5;
rad = floor(dilationAmount);
[r,c] = size(Thyroid);
filledImage = imfill(Thyroid, 'holes');

for i=1:r
   for j=1:c
       x1=i-rad;
       x2=i+rad;
       y1=j-rad;
       y2=j+rad;
       if x1<1
           x1=1;
       end
       if x2>r
           x2=r;
       end
       if y1<1
           y1=1;
       end
       if y2>c
           y2=c;
       end
       erodedImage(i,j) = min(min(filledImage(x1:x2,y1:y2)));
   end
end
ThyroidOutline=Thyroid;
ThyroidOutline(erodedImage)=0;
rgb = inp(:,:,[1 1 1]);
red = rgb(:,:,1);
red(ThyroidOutline)=255;
green = rgb(:,:,2);
green(ThyroidOutline)=0;
blue = rgb(:,:,3);
blue(ThyroidOutline)=0;
ThyroidOutlineInserted(:,:,1) = red; 
ThyroidOutlineInserted(:,:,2) = green; 
ThyroidOutlineInserted(:,:,3) = blue; 
axes(handles.axes3); imshow(inp); title('Segmentation');
hold on;rectangle('Position',wantedBox,'EdgeColor','y');hold off;
